#' Saves a model as a `.keras` file.
#'
#' @description
#'
#' # Examples
#' ```python
#' model = keras.Sequential(
#'     [
#'         keras.layers.Dense(5, input_shape=(3,)),
#'         keras.layers.Softmax(),
#'     ],
#' )
#' model.save("model.keras")
#' loaded_model = keras.saving.load_model("model.keras")
#' x = keras.random.uniform((10, 3))
#' assert np.allclose(model.predict(x), loaded_model.predict(x))
#' ```
#'
#' Note that `model.save()` is an alias for `keras.saving.save_model()`.
#'
#' The saved `.keras` file contains:
#'
#' - The model's configuration (architecture)
#' - The model's weights
#' - The model's optimizer's state (if any)
#'
#' Thus models can be reinstantiated in the exact same state.
#'
#' @param model
#' Keras model instance to be saved.
#'
#' @param filepath
#' `str` or `pathlib.Path` object. Path where to save the model.
#'
#' @param overwrite
#' Whether we should overwrite any existing model at the target
#' location, or instead ask the user via an interactive prompt.
#'
#' @param ...
#' For forward/backward compatability.
#'
#' @export
#' @family saving
#' @seealso
#' + <https:/keras.io/keras_core/api/models/model_saving_apis/model_saving_and_loading#savemodel-function>
#' + <https://www.tensorflow.org/api_docs/python/tf/keras/saving/save_model>
saving_save_model <-
__signature__
keras.saving.save_model(
  model,
  filepath,
  overwrite=True,
  **kwargs
)
